**Greatest of 3 numbers** - 

*Code* - [2]

#include<stdio.h>
int main()
{	
	int a,b,c,max;

	printf("Enter 3 numbers\n");
	scanf("%d%d%d",&a,&b,&c);

	if(a>b&&a>c)
		max=a;
	else if(b>c)
		max=b;
	else
		max=c;

	printf("Max = %d\n",max);

	return 0;
}
