**Sum of 1st and 3rd digits of a 5-digit +ve number**  - [2]

```
#include<stdio.h>
int main()
{	
	int num,sum=0;

	printf("Enter a 5-digit positive number\n");
	scanf("%d",&num);

	num/=100;
	sum+=num%10;
	num/=100;
	sum+=num%10;

	printf("Sum = %d\n",sum);

	return 0;
}
```  
